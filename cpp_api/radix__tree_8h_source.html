576  reflog *&get_or_create(const key_t &key, ARGS &&... args) {
577  node_t *node = root_;
578  for (size_t d = 0; d < depth_ - 1; d++) {
579  node_t *child = nullptr;
580  if ((child = atomic::load(&(node->children()[key[d]]))) == nullptr) {
581  // Try & allocate child node
582  void *raw = allocator::instance().alloc(sizeof(node_t));
583  child = new(raw) node_t(key[d], width_, d + 1, node);
584  node_t *expected = nullptr;
585 
586  // If thread was not successful in swapping newly allocated memory,
587  // then it should de-allocate memory, and accept whatever the
588  // successful thread allocated as the de-facto storage for child node.
589  if (!atomic::strong::cas(&(node->children()[key[d]]), &expected, child)) {
590  child->~node_t();
591  allocator::instance().dealloc(child);
592  child = expected;
593  }
594  }
595  // child is definitely allocated now
596  node = child;
597  }
598 
599  // Reached leaf
600  size_t d = depth_ - 1;
601  node_t *child = nullptr;
602  if ((child = atomic::load(&(node->children()[key[d]]))) == nullptr) {
603  // Try & allocate child node
604  void *raw = allocator::instance().alloc(sizeof(node_t));
605  child = new(raw) node_t(key[d], width_, d + 1, node, true, std::forward<ARGS>(args)...);
606  node_t *expected = nullptr;
607 
608  // If thread was not successful in swapping newly allocated memory,
609  // then it should de-allocate memory, and accept whatever the
610  // successful thread allocated as the de-facto storage for child node.
611  if (!atomic::strong::cas(&(node->children()[key[d]]), &expected, child)) {
