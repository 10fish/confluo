327  __atomic_bucket_ref *try_allocate_container(size_t container_idx) {
328  size_t num_buckets = 1U << (container_idx + FCB_HIBIT);
329  __atomic_bucket_ref *new_container = new __atomic_bucket_ref[num_buckets]();
330  __atomic_bucket_ref *expected = nullptr;

332  // Only one thread will be successful in replacing the nullptr reference with newly allocated container.
333  if (!atomic::strong::cas(&bucket_containers_[container_idx], &expected, new_container)) {
334  // All other threads will deallocate the newly allocated container.
335  delete[] new_container;
336  return expected;
