325  void atomic_copy(read_only_encoded_ptr<T> &copy, size_t offset = 0) const {
326  // Increment both counters to guarantee that the loaded
327  // pointer can't be deallocated if there are no copies.
328  // Protects against loading before a swap begins and
329  // making a copy after the swap finishes.
330  ref_counts_.increment_both();
331  encoded_ptr<T> ptr = atomic::load(&enc_ptr_);
332 
333  if (ptr.ptr() == nullptr) {
334  // Decrement both ref counts (no possibility of them reaching 0 here)
335  ref_counts_.decrement_both();
336  return;
337  }
338 
339  // Correct the other counter and create the copy.
340  auto aux = ptr_aux_block::get(ptr_metadata::get(ptr.ptr()));
341  if (aux.state_ == state_type::D_IN_MEMORY) {
342  // decrement other ref count (no possibility of it reaching 0 here)
343  ref_counts_.decrement_second();
344  copy.init(ptr, offset, &ref_counts_);
345  return;
346  } else if (aux.state_ == state_type::D_ARCHIVED) {
347  // decrement other ref count (no possibility of it reaching 0 here)
348  ref_counts_.decrement_first();
349  copy.init(ptr, offset, &ref_counts_);
350  return;
351  } else {
352  THROW(memory_exception, "Unsupported pointer state during copy!");
353  }
354  }
