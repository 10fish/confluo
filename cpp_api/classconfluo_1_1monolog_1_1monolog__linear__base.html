template<typename T, size_t MAX_BUCKETS = 4096, size_t BUCKET_SIZE = 268435456, size_t BUFFER_SIZE = 1048576>
class confluo::monolog::monolog_linear_base< T, MAX_BUCKETS, BUCKET_SIZE, BUFFER_SIZE >

Monolog linear base class. The monolog grow linearly, bucket-by-bucket, as space runs out in previous blocks.
