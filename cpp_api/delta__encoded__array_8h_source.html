205  elias_gamma_prefix_sum() {
206  for (uint64_t i = 0; i < 65536; i++) {
207  uint16_t val = (uint16_t) i;
208  uint64_t count = 0, offset = 0, sum = 0;
209  while (val && offset <= 16) {
210  int N = 0;
211  while (!GETBIT(val, offset)) {
212  N++;
213  offset++;
214  }
215  offset++;
216  if (offset + N <= 16) {
217  sum += ((val >> offset) & ((uint16_t) low_bits_set[N])) + (1 << N);
218  offset += N;
219  count++;
220  } else {
221  offset -= (N + 1);
222  break;
223  }
224  }
225  prefixsum_[i] = static_cast<uint32_t>((offset << 24) | (count << 16) | sum);
226  }
227  }

505  T prefix_sum(pos_type delta_offset, pos_type until_idx) {
506  T delta_sum = 0;
507  pos_type delta_idx = 0;
508  pos_type current_delta_offset = delta_offset;
509  while (delta_idx != until_idx) {
510  uint16_t block = this->deltas_->template get_val_pos<uint16_t>(
511  current_delta_offset, 16);
512  uint16_t cnt = elias_gamma_prefix_table.count(block);
513  if (cnt == 0) {
514  // If the prefixsum table for the block returns count == 0
515  // this must mean the value spans more than 16 bits
516  // read this manually
517  width_type delta_width = 0;
518  while (!this->deltas_->get_bit(current_delta_offset)) {
519  delta_width++;
520  current_delta_offset++;
521  }
522  current_delta_offset++;
523  delta_sum += this->deltas_->template get_val_pos<T>(
524  current_delta_offset, delta_width) + (1ULL << delta_width);
525  current_delta_offset += delta_width;
526  delta_idx += 1;
527  } else if (delta_idx + cnt <= until_idx) {
528  // If sum can be computed from the prefixsum table
529  delta_sum += elias_gamma_prefix_table.sum(block);
530  current_delta_offset += elias_gamma_prefix_table.offset(block);
531  delta_idx += cnt;
532  } else {
533  // Last few values, decode them without looking up table
534  while (delta_idx != until_idx) {
535  width_type delta_width = 0;
536  while (!this->deltas_->get_bit(current_delta_offset)) {
537  delta_width++;
538  current_delta_offset++;
539  }
540  current_delta_offset++;
541  delta_sum += this->deltas_->template get_val_pos<T>(
542  current_delta_offset, delta_width) + (1ULL << delta_width);
543  current_delta_offset += delta_width;
544  delta_idx += 1;
545  }
546  }
547  }
