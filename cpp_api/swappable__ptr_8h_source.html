254  void atomic_copy(read_only_ptr<T> &copy, size_t offset = 0) const {
255  // Increment both counters to guarantee that the loaded
256  // pointer can't be deallocated if there are no copies.
257  // Protects against loading before a swap begins and
258  // making a copy after the swap finishes.
259  ref_counts_.increment_both();
260  T *ptr = atomic::load(&ptr_);
261 
262  if (ptr == nullptr) {
263  // Decrement both ref counts (no possibility of them reaching 0 here)
264  ref_counts_.decrement_both();
265  return;
266  }
267 
268  // Correct the other counter and create the copy.
269  auto aux = ptr_aux_block::get(ptr_metadata::get(ptr));
270  if (aux.state_ == state_type::D_IN_MEMORY) {
271  // decrement other ref count (no possibility of it reaching 0 here)
272  ref_counts_.decrement_second();
273  copy.init(ptr, offset, &ref_counts_);
274  return;
275  } else if (aux.state_ == state_type::D_ARCHIVED) {
276  // decrement other ref count (no possibility of it reaching 0 here)
277  ref_counts_.decrement_first();
278  copy.init(ptr, offset, &ref_counts_);
279  return;
